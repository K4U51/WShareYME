# LVGL Image Mapping Guide
# ------------------------
# Original PNG file → Converted C file → LVGL Symbol Name → Usage Example

g_img1.png    → g_img1.c    → ui_img_g_img1_png    → lv_img_set_src(g_img1, &ui_img_g_img1_png);
g_img2.png    → g_img2.c    → ui_img_g_img2_png    → lv_img_set_src(g_img2, &ui_img_g_img2_png);
g_img3.png    → g_img3.c    → ui_img_g_img3_png    → lv_img_set_src(g_img3, &ui_img_g_img3_png);

# Optional descriptive names (more readable)
bg1.png       → bg1.c       → ui_img_bg1_png       → lv_img_set_src(g_img1, &ui_img_bg1_png);
bg2.png       → bg2.c       → ui_img_bg2_png       → lv_img_set_src(g_img2, &ui_img_bg2_png);
overlay.png   → overlay.c   → ui_img_overlay_png   → lv_img_set_src(g_img3, &ui_img_overlay_png);

# Notes:
# 1. LVGL symbol names are derived from the PNG filename, prefixed with 'ui_img_' and suffixed with '_png'.
# 2. The .c file generated by the converter may match the original PNG name but you always use the symbol in code.
# 3. Keep filenames lowercase and without spaces for safe conversion.
# 4. Use lv_img_set_src() with the & symbol pointing to the declared LVGL image.
